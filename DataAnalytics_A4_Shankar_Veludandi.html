  Tax_Total = "Tax Total",
  Tax_Threshold = "Tax Threshold",
  Total_Cash = "Total Cash",
  Total_AAV = "Total AAV",
  Offseason_Total = "Offseason Total"
)

# Remove $ and commas and convert financial columns to numeric
financial_cols <- c("Cap_Space","Cap_Max","Tax_Total","Tax_Threshold",
                    "Total_Cash","Total_AAV","Offseason_Total")
df <- df %>% mutate(across(all_of(financial_cols), ~ parse_number(.)))

# Generate financial utilization metrics
df <- df %>% mutate(
  cap_util = (Cap_Max - Cap_Space) / Cap_Max,
  tax_util = Tax_Total / Tax_Threshold,
  cash_util = Total_Cash / Tax_Threshold,
  aav_util  = Total_AAV / Cap_Max,
  offspend_util = Offseason_Total / Cap_Max
)

# 2. Filter to regular-season data and compute PCA on performance metrics

4. Build Team-Year Panel and Lag Predictors

# 4. Create panel data with lagged financial metrics and next-year outcomes
# Add a numeric season identifier (season_start year) for sorting
df_reg <- df_reg %>% mutate(season_start = as.integer(substr(Season, 1, 4)))

# Arrange by team and season, then create lags and leads
panel_data <- df_reg %>% arrange(TEAM_ID, season_start) %>%
  group_by(TEAM_ID) %>%
  mutate(
    cap_util_lag   = lag(cap_util, 1),
    tax_util_lag   = lag(tax_util, 1),
    cash_util_lag  = lag(cash_util, 1),
    aav_util_lag   = lag(aav_util, 1),
    offspend_lag   = lag(offspend_util, 1),
    perf_pc1_nx    = lead(perf_pc1, 1),
    playoff_nx     = lead(playoff, 1)
  ) %>%
  ungroup()

# Remove seasons with missing lag or next-year outcome (e.g., first and last season for each team)
panel_data <- panel_data %>% 
  filter(!is.na(cap_util_lag) & !is.na(perf_pc1_nx))

6. Predictive Modeling

6.1 Regression Models for Next-Year Performance

# 6.1 Regression models for perf_pc1_nx
# Define regression formula using lagged financial predictors
reg_formula <- perf_pc1_nx ~ cap_util_lag + tax_util_lag + cash_util_lag + aav_util_lag + offspend_lag

# **(a) Pooled OLS Regression**
ols_model <- lm(reg_formula, data=train_data)

# **(b) Fixed Effects Regression (team fixed effects)**
# Include team as a factor to account for team-specific intercepts
fe_model <- lm(perf_pc1_nx ~ 0 + factor(TEAM_ID) + cap_util_lag + tax_util_lag + 
               cash_util_lag + aav_util_lag + offspend_lag, data=train_data)
# Note: "0 + factor(TEAM_ID)" removes the global intercept and adds an indicator for each team (fixed effects).

# **(c) LASSO Regression** (with cross-validation for lambda)
