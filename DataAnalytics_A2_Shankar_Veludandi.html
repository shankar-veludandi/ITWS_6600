3) Train 2 kNN models using region as the class label

3.1

region1 <- "Southern Asia"
region2 <- "Asia-Pacific"

df_knn <- EPI_data %>%
  filter(region %in% c(region1, region2)) %>%
  select(region, EPI.old, EPI.new, ECO.old)  

df_knn$region <- factor(df_knn$region)

df_knn$EPI.old <- as.numeric(df_knn$EPI.old)
df_knn$EPI.new <- as.numeric(df_knn$EPI.new)
df_knn$ECO.old <- as.numeric(df_knn$ECO.old)

set.seed(123)
trainIndex <- createDataPartition(df_knn$region, p = 0.7, list = FALSE)
trainData <- df_knn[trainIndex, ]
testData  <- df_knn[-trainIndex, ]

trainX <- trainData[, c("EPI.old", "EPI.new", "ECO.old")]
trainY <- trainData$region
testX  <- testData[, c("EPI.old", "EPI.new", "ECO.old")]
testY  <- testData$region

knn_pred <- knn(train = trainX,
                test  = testX,
                cl    = trainY,
                k     = 3)

confusion <- table(Predicted = knn_pred, Actual = testY)
print(confusion)
##                Actual
## Predicted       Asia-Pacific Southern Asia
##   Asia-Pacific             6             2
##   Southern Asia            1             0
accuracy <- sum(diag(confusion)) / sum(confusion)
cat("Accuracy for k=3:", accuracy, "\n")
## Accuracy for k=3: 0.6666667
for (k in c(1, 5, 7)) {
  knn_pred_k <- knn(trainX, testX, cl = trainY, k = k)
  confusion_k <- table(Predicted = knn_pred_k, Actual = testY)
  accuracy_k <- sum(diag(confusion_k)) / sum(confusion_k)
  cat("k =", k, " Accuracy =", accuracy_k, "\n")
}

                cl    = trainY,
                k     = 3)

##                Actual

##   Southern Asia            1             0
accuracy <- sum(diag(confusion)) / sum(confusion)
cat("Accuracy for k=3:", accuracy, "\n")
## Accuracy for k=3: 0.6666667
